00934 private: System::Void b_Process_Click(System::Object^  sender, System::EventArgs^  e) {
00935                          float sSize;

00951                          if (host->getOk() && (combo_Type->SelectedIndex!=-1) && sSizeOk) {     //ak su zvolene vsetky parametre programu
00952                                  int* selectedParams = new int[numParams];

00961                                                  volType[numSelectedParams] = ((System::Windows::Forms::CheckBox^)panel_Vol->Controls[i])->Checked;

00964                                  }
00965

00967                                  host->setProcessed(false);
00968                                  if (numSelectedParams>0) {
00969                                         console->AppendText("Processing...");
00970                                         progressBar1->Value = 0;

00976                                          console->AppendText("ERROR: no parameter selected" + nl);      
00977                                  }
00978                                  delete [] stepSize;
00979                                  delete [] volType;
00980                                  delete [] selectedParams;
